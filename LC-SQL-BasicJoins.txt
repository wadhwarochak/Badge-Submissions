1378. Replace Employee ID With The Unique Identifier

select eu.unique_id, e.name
from Employees e
left join EmployeeUNI eu
on e.id = eu.id;

SELECT product_name, s.year,s.price
from sales s
join product p
on p.product_id = s.product_id;
